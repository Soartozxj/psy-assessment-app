## ✅ 已完成的优化

### **优化1：异步加载 `shared-data.js`** ✅

**修改前**:

```html
<script src="shared-data.js?v=3"></script>
```

**修改后**:

```javascript
function loadScriptAsync(src, onLoad) {
  var script = document.createElement('script');
  script.src = src;
  script.onload = onLoad || function () {};
  script.onerror = function () {
    console.warn('[Admin] 加载失败: ' + src);
    if (onLoad) onLoad();
  };
  document.head.appendChild(script);
}

// 按依赖顺序异步加载
loadScriptAsync('shared-data.js?v=3', function () {
  loadScriptAsync('cloud-data.js', function () {
    loadScriptAsync('data-monitor.js', function () {
      console.log('[Admin] 大型脚本异步加载完成');
    });
  });
});
```

**优化效果**:

- ✅ **消除页面阻塞**（`shared-data.js` 725KB 异步加载）
- ✅ **提升首屏速度** (页面可以立即显示)
- ✅ **按依赖顺序加载** (`cloud-data.js` 等 `shared-data.js` 完成后再加载)

### **优化3：异步加载依赖脚本** ✅

**修改前**:

```html
<!-- 运维手册数据模块 (P1-2) -->
<script src="ops-manual-data.js?v=1"></script>
<!-- 量表编辑分步向导模块 (P1-4) -->
<script src="admin-scale-wizard.js?v=1"></script>
</head>
```

**修改后**:

```javascript
<!-- 运维手册数据模块 (P1-2) 和 量表编辑分步向导模块 (P1-4) -->
<!-- 改为异步加载，等待 shared-data.js 加载完成后执行 -->
<script>
  // 等待大型脚本加载完成后再加载依赖脚本
  function loadDependentScripts() {
    if (typeof SharedData !== 'undefined') {
      // shared-data.js 已加载，可以加载依赖它的脚本
      loadScriptAsync('ops-manual-data.js?v=1', function() {
        loadScriptAsync('admin-scale-wizard.js?v=1', function() {
          console.log('[Admin] 依赖脚本加载完成');
        });
      });
    } else {
      // 等待 100ms 后重试
      setTimeout(loadDependentScripts, 100);
    }
  }

  // 在 shared-data.js 加载完成后调用
  const originalOnLoad = loadScriptAsync.toString();
  setTimeout(loadDependentScripts, 3000); // 3秒后尝试加载（此时 shared-data.js 应该已加载完成）
</script>
```

**优化效果**:

- ✅ **解决依赖问题** (等待 `shared-data.js` 加载完成后再加载)
- ✅ **避免执行顺序错误** (`admin-scale-wizard.js` 不会再找不到 `SharedData`)
- ✅ **自动重试** (100ms 后重试，直到 `SharedData` 可用)

### **优化4：修复插件路径问题** ✅

**问题**:

```
GET http://localhost:8081/mini-app-h5/backend/plugins/optional/ai-plugin.js 404 (File not found)
```

**原因**:

- `ai-plugin.js` 等文件在 `core/` 目录
- 但 `plugin-loader.js` 将它们识别为 `optional` 插件
- 去 `optional/` 目录查找，找不到

**解决方案**:
将 `ai-plugin.js`、`scale-plugin.js`、`scoring-plugin.js`、`npc-plugin.js` 复制到 `optional/` 目录

**执行命令**:

```bash
cd /Users/rich/WorkBuddy/20260407113106/mini-app-h5/backend/plugins
cp core/ai-plugin.js optional/
cp core/scale-plugin.js optional/
cp core/scoring-plugin.js optional/
cp core/npc-plugin.js optional/
```

**优化效果**:

- ✅ **解决404错误** (插件文件在 `optional/` 目录也能找到)
- ✅ **支持按需加载** (`ai`、`scale` 等插件可以正常按需加载)

```javascript
// 加载进度管理器
window.LoadingManager = {
  overlay: null,
  text: null,
  progressBar: null,
  currentStep: 0,
  totalSteps: 4,

  init() {
    this.overlay = document.getElementById('loading-overlay');
    this.text = document.getElementById('loading-text');
    this.progressBar = document.getElementById('loading-progress-bar');
  },

  update(step, message) {
    this.currentStep = step;
    const percent = Math.min((step / this.totalSteps) * 100, 100);
    if (this.progressBar) {
      this.progressBar.style.width = percent + '%';
    }
    if (this.text && message) {
      this.text.textContent = message;
    }
  },

  hide() {
    if (this.overlay) {
      this.overlay.classList.add('fade-out');
      setTimeout(() => {
        this.overlay.style.display = 'none';
      }, 500);
    }
  }
};

document.addEventListener('DOMContentLoaded', async function () {
  // 初始化加载管理器
  window.LoadingManager.init();
  window.LoadingManager.update(0, '正在初始化...');

  console.log('🚀 开始初始化插件系统...');

  try {
    // 1. 只预加载 auth 插件（认证必需）
    window.LoadingManager.update(1, '正在加载认证模块...');

    // 其他插件（ai, scale, scoring, npc）改为按需加载
    await PluginLoader.init();

    console.log('✅ 插件系统初始化完成');
    console.log('   已加载插件:', Object.keys(PluginLoader.plugins).join(', ') || '（无，等待按需加载）');

    // 2. 插件加载完成后，执行认证检查
    window.LoadingManager.update(2, '正在检查认证...');

    if (typeof adminAuth !== 'undefined' && adminAuth.init) {
      console.log('🔐 开始认证检查...');
      adminAuth.init();

      // 3. 如果已认证，初始化页面
      if (Adapter.storage.getSession('psy_admin_authed') === '1') {
        window._adminInitialized = true;
        window.LoadingManager.update(3, '正在加载页面...');
        switchSection('dashboard');

        // 4. 隐藏加载指示器
        setTimeout(() => {
          window.LoadingManager.update(4, '加载完成！');
          window.LoadingManager.hide();
        }, 500);

        // 5. 后台预加载其他常用插件（不阻塞页面）
        setTimeout(() => {
          console.log('🔄 后台预加载常用插件...');
          PluginLoader.load('ai', true).catch(() => {});
          PluginLoader.load('scale', true).catch(() => {});
        }, 2000); // 2秒后预加载
      } else {
        // 未认证，隐藏加载指示器，显示登录界面
        window.LoadingManager.hide();
      }
    } else {
      console.error('❌ adminAuth 未定义，无法执行认证检查');
      window.LoadingManager.hide();
    }
  } catch (error) {
    console.error('❌ 插件系统初始化失败:', error);
    window.LoadingManager.update(0, '加载失败，请刷新页面');
    setTimeout(() => window.LoadingManager.hide(), 3000);
  }
});
```

**优化效果**:

- ✅ **提升用户体验** (用户可以看到加载进度)
- ✅ **减少重复刷新** (用户知道页面在加载)
- ✅ **专业感** (渐变背景 + 旋转动画 + 进度条)
