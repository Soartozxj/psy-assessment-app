### FIX-001: 修复 eval() 使用

**风险等级**: 🔴 高风险  
**影响文件**:

- `mini-app-h5/backend/scoring-engine.js`
- `tests/test-plugins.js`

**问题描述**:  
使用 `eval()` 执行动态生成的代码，如果输入可控，会导致代码注入攻击。

**修复方案**:

#### 方案1: 使用 Function 构造函数（推荐）

**优点**:

- 比 `eval()` 安全，不直接访问局部作用域
- 可以明确指定参数，限制可执行代码的作用域

**实施步骤**:

1. **修改 scoring-engine.js**

```javascript
// 原代码 (第~145行)
function evaluateDerived(derivedMetrics, currentValues) {
  // 按依赖顺序排列，替换指标名后 eval
  const code = generateCode(derivedMetrics);
  return eval(code);
}

// 修复后
function evaluateDerived(derivedMetrics, currentValues) {
  // 按依赖顺序排列，替换指标名后使用 Function 构造函数
  const code = generateCode(derivedMetrics);

  // 白名单验证：确保代码只包含允许的字符
  if (!isSafeExpression(code)) {
    throw new Error('不安全的表达式');
  }

  // 使用 Function 构造函数，明确参数
  const fn = new Function('metricValues', 'return ' + code);
  return fn(currentValues);
}

// 添加安全验证函数
function isSafeExpression(expr) {
  // 只允许字母、数字、下划线、点、运算符、括号、空格
  return /^[a-zA-Z0-9_\.\+\-\*\/\(\)\s]+$/.test(expr);
}
```

2. **修改 test-plugins.js**（测试环境）

```javascript

// 修复后（使用 vm 模块，仅测试环境）
```

```javascript
// 添加验证中间件
const validate = (validations) => {

    await Promise.all(validations.map((validation) => validation.run(req)));
```

```javascript
  return input.replace(/<[^>]*>/g, '').replace(/[<>'"&]/g, (char) => {
```

```javascript
const express = require('express');
const app = express();

// 强制 HTTPS（生产环境）
if (process.env.NODE_ENV === 'production') {
  app.use((req, res, next) => {
    // 检查是否是 HTTPS 或经过代理的 HTTPS
    if (req.secure || req.headers['x-forwarded-proto'] === 'https') {
      next();
    } else {
      res.redirect(`https://${req.headers.host}${req.url}`);
    }
  });
}

// 添加安全 Headers
app.use((req, res, next) => {
  // 强制 HTTPS（HSTS）
  res.setHeader('Strict-Transport-Security', 'max-age=31536000; includeSubDomains; preload');

  // 防止 MIME 类型嗅探
  res.setHeader('X-Content-Type-Options', 'nosniff');

  // 防止点击劫持
  res.setHeader('X-Frame-Options', 'SAMEORIGIN');

  // XSS 保护
  res.setHeader('X-XSS-Protection', '1; mode=block');

  next();
});

app.listen(3000, () => {
  console.log('Server running on port 3000');
});
```

```javascript
// 在 auth-plugin.js 中
class AuthPlugin extends PluginBase {
  // ... 现有代码 ...

  hasPermission(user, permission) {
    if (!user || !user.role) return false;

    const userPermissions = PERMISSIONS[user.role];
    if (!userPermissions) return false;

    return userPermissions.includes(permission);
  }

  requirePermission(permission) {
    return (req, res, next) => {
      const user = req.user; // 从 session/token 获取

      if (!this.hasPermission(user, permission)) {
        return res.status(403).json({
          success: false,
          error: '权限不足'
        });
      }

      next();
    };
  }
}
```

### FIX-006: CSRF 保护

**风险等级**: 🟡 中风险

**实施步骤**:

1. **安装依赖**

2. **修改 server/psy-api.js**

```javascript
const csurf = require('csurf');
const cookieParser = require('cookie-parser');

app.use(cookieParser());
app.use(csurf({ cookie: true }));

// 提供 CSRF Token 给前端
app.get('/api/csrf-token', (req, res) => {
  res.json({ csrfToken: req.csrfToken() });
});

// 错误处理
app.use((err, req, res, next) => {
  if (err.code === 'EBADCSRFTOKEN') {
    res.status(403).json({
      success: false,
      error: 'CSRF Token 无效'
    });
  } else {
    next(err);
  }
});
```

3. **前端获取并使用 CSRF Token**

```javascript
// 在 admin-legacy.html 中
let csrfToken = '';

// 页面加载时获取 CSRF Token
async function initCSRF() {
  const response = await fetch('/api/csrf-token', {
    credentials: 'include'
  });
  const data = await response.json();
  csrfToken = data.csrfToken;

  // 设置到 cookie
  document.cookie = `csrfToken=${csrfToken}; SameSite=Strict`;
}

// 在所有 API 请求中包含 CSRF Token
async function apiRequest(url, options = {}) {
  const defaultOptions = {
    credentials: 'include',
    headers: {
      'X-CSRF-Token': csrfToken,
      'Content-Type': 'application/json',
      ...options.headers
    }
  };

  const response = await fetch(url, { ...options, ...defaultOptions });
  return response.json();
}

// 页面加载时初始化
document.addEventListener('DOMContentLoaded', initCSRF);
```
